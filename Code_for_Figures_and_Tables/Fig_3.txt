import pandas as pd
import numpy as np
from scipy.spatial.distance import pdist, squareform
from sklearn.decomposition import PCA
import matplotlib.pyplot as plt

# File paths
otu_table_path = "/Users/christopherbivins/Desktop/Oak_Endophyte_Project_NEATEST/Abundance_CSV_Files/Relative_Abundance/RA_otu_table.csv"
metadata_path = "/Users/christopherbivins/Desktop/Oak_Endophyte_Project_NEATEST/Abundance_CSV_Files/Relative_Abundance/metadata.csv"

# Load data
otu_table = pd.read_csv(otu_table_path, index_col=0)
metadata = pd.read_csv(metadata_path)

# Align metadata with OTU table
metadata = metadata[metadata["Sample_ID"].isin(otu_table.columns)].set_index("Sample_ID")

# Transpose OTU table for distance calculation (samples as rows)
otu_table_transposed = otu_table.T

# Compute Bray-Curtis distance matrix
bray_curtis_distances = pdist(otu_table_transposed, metric="braycurtis")
bray_curtis_matrix = squareform(bray_curtis_distances)

# Save Bray-Curtis distance matrix
pd.DataFrame(bray_curtis_matrix, index=otu_table_transposed.index, columns=otu_table_transposed.index).to_csv("bray_curtis_distance_matrix.csv")

# Principal Coordinates Analysis (PCoA)
pcoa = PCA(n_components=3)
pcoa_result = pcoa.fit_transform(bray_curtis_matrix)
variance_explained = pcoa.explained_variance_ratio_ * 100

# Save PCoA results
pcoa_df = pd.DataFrame(pcoa_result, columns=["Axis 1", "Axis 2", "Axis 3"], index=otu_table_transposed.index)
pcoa_df.to_csv("pcoa_results.csv")

# Define color palettes
species_palette = {"Q_douglasii": "#88CCEE", "Q_wislizeni": "#44AA99"}
month_palette = {"May": "#E69F00", "July": "#56B4E9", "September": "#009E73"}
site_palette = {"Ridge": "#D55E00", "Valley": "#0072B2"}

# Function to plot PCoA with legends
def plot_pcoa(color_by, palette, filename):
    plt.figure(figsize=(10, 8))
    unique_labels = metadata[color_by].unique()
    handles = []
    
    for sample in pcoa_df.index:
        if sample in metadata.index:
            color = palette[metadata.loc[sample, color_by]]
            scatter = plt.scatter(
                pcoa_df.loc[sample, "Axis 1"],
                pcoa_df.loc[sample, "Axis 2"],
                color=color, edgecolors='black', alpha=0.8, s=100
            )
            
    for label in unique_labels:
        handles.append(plt.Line2D([0], [0], marker='o', color='w', markerfacecolor=palette[label], markersize=10, label=label))
    
    plt.xlabel(f"PCoA Axis 1 ({variance_explained[0]:.2f}%)")
    plt.ylabel(f"PCoA Axis 2 ({variance_explained[1]:.2f}%)")
    plt.title(f"PCoA Plot Colored by {color_by}")
    plt.legend(handles=handles, title=color_by)
    plt.grid(False)
    plt.savefig(filename, format="svg")
    plt.show()

# Generate and save plots
plot_pcoa("Species", species_palette, "pcoa_species.svg")
plot_pcoa("Month", month_palette, "pcoa_month.svg")
plot_pcoa("Site", site_palette, "pcoa_site.svg")

```{r}
# ============================================================================
# varpart_analysis.R
# Purpose: Partition variance in fungal endophyte community composition among
#          Species, Site, and Month using distance-based redundancy analysis.
#          Complements the PERMANOVA (adonis2) results by separating unique
#          and shared variance fractions for each factor.
# Inputs:  RA_otu_table.csv (relative abundance OTU table, OTUs as rows, samples as cols)
#          metadata.csv (sample metadata with Species, Site, Month, TreeID)
# Logic:   Uses vegan::varpart() with Bray-Curtis dissimilarities via capscale
#          (db-RDA) to partition variance into unique and shared components
#          for three explanatory variable sets: Species, Site, and Month.
# ============================================================================

library(vegan)

# ---------------------------
# Set file paths (EDIT THESE)
# ---------------------------
otu_table_path <- "/Users/cbivins/Desktop/Finished_Projects/Oak_Endophyte_Project_NEATEST/NEW/RA_otu_table.csv"
metadata_path  <- "/Users/cbivins/Desktop/Finished_Projects/Oak_Endophyte_Project_NEATEST/NEW/metadata.csv"
output_dir     <- "/Users/cbivins/Desktop/Finished_Projects/Oak_Endophyte_Project_NEATEST/NEW/Varpart"

# ---------------------------
# Load and align data
# ---------------------------

# Load OTU table (OTUs as rows, samples as columns)
otu_table <- read.csv(otu_table_path, row.names = 1)

# Load metadata
metadata <- read.csv(metadata_path)

# Standardize Sample_IDs to match OTU table column names (hyphens to dots)
metadata$Sample_ID <- gsub("-", ".", metadata$Sample_ID)

# Subset metadata to only samples present in OTU table
metadata <- metadata[metadata$Sample_ID %in% colnames(otu_table), ]

# Order metadata to match OTU table column order
metadata <- metadata[match(colnames(otu_table), metadata$Sample_ID), ]

# Subset OTU table to only samples present in metadata
shared_samples <- intersect(colnames(otu_table), metadata$Sample_ID)
otu_table <- otu_table[, shared_samples]
metadata <- metadata[match(shared_samples, metadata$Sample_ID), ]

# Verify alignment
stopifnot(all(metadata$Sample_ID == colnames(otu_table)))

# Transpose OTU table so samples are rows (required by vegan)
otu_t <- as.data.frame(t(otu_table))

# ---------------------------
# Compute Bray-Curtis distance matrix
# ---------------------------
bray_curtis <- vegdist(otu_t, method = "bray")

# ---------------------------
# Create explanatory variable matrices
# ---------------------------
# varpart needs each predictor set as a separate data frame or matrix
# For categorical variables, we pass them as factors in data frames

species_df <- data.frame(Species = factor(metadata$Species))
site_df    <- data.frame(Site = factor(metadata$Site))
month_df   <- data.frame(Month = factor(metadata$Month))

# ---------------------------
# Run variance partitioning
# ---------------------------
# varpart with a distance matrix uses db-RDA (capscale) internally
# Three explanatory tables: Species, Site, Month
vp <- varpart(bray_curtis, species_df, site_df, month_df)

# Print full results to console
cat("\n========================================\n")
cat("Variance Partitioning Results\n")
cat("========================================\n\n")
print(vp)

# ---------------------------
# Extract and export results table
# ---------------------------
frac_labels <- c(
  "[a] Species alone",
  "[b] Site alone",
  "[c] Month alone",
  "[d] Species + Site shared",
  "[e] Species + Month shared",
  "[f] Site + Month shared",
  "[g] All three shared",
  "Residual"
)

results_table <- data.frame(
  Fraction    = frac_labels,
  Adj.R.sq    = round(frac_table$Adj.R.square, 4),
  Testable    = frac_table$Testable
)

print(results_table, row.names = FALSE)

write.csv(results_table, file.path(output_dir, "varpart_results.csv"), row.names = FALSE)
